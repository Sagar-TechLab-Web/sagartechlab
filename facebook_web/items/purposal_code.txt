<script>
    const noBtn = document.getElementById("noBtn");
    const card = document.getElementById("mainCard");

    function moveAnywhere(){
        const maxX = card.clientWidth - noBtn.offsetWidth;
        const maxY = card.clientHeight - noBtn.offsetHeight;

        const x = Math.random() * maxX;
        const y = Math.random() * (maxY - 100) + 100;

        noBtn.style.left = x + "px";
        noBtn.style.top = y + "px";
    }

    noBtn.addEventListener("mouseover", moveAnywhere);
    noBtn.addEventListener("mousemove", moveAnywhere);
    noBtn.addEventListener("click", moveAnywhere);

    function yesClicked(){
        document.getElementById("yesSound").play();
        card.style.display = "none";
        document.getElementById("successCard").style.display = "block";
    }
</script>
